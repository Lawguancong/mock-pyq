# 本地 run
1. yarn install
2. yarn build 
3. yarn server (port: 4001)
4. yarn start
5. visit localhost: 4002

# 线上 run
1. yarn install
2. pm2 start server/index.config.js 
3. yarn start
4. visit localhost:4002 // localhost:4001
